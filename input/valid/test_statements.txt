void main() {
    int[] array = { 23, 12903, 53, -128, 0, 24 };

    int n = array.length;
    int temp = 0;
    int i = 0;

    # sort array
    while (i < n) {
        int j = 1;

        while (j < n-i) {
            if (array[j-1] > array[j]) {
                temp = array[j-1];
                array[j-1] = array[j];
                array[j] = temp;
            }

            j = j + 1;
        }

        i = i + 1;
    }
}
